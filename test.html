// SkyWay APIキーを設定 const peer = new Peer({ key: "e5b9d8b9-ea5a-4363-88e5-c6a752fbab07", debug: 3 }); document.getElementById('start-call-button').onclick = async () => { // 通話部屋に入る const room = peer.joinRoom("test", { mode: 'sfu' });//p2p // 通話が開始されたときの処理 room.on('open', () => { console.log('通話が開始されました'); }); // 相手からの音声が届いたときの処理 room.on('stream', stream => { const audio = document.createElement('audio'); audio.srcObject = stream; audio.play(); }); document.getElementById('start-call-button').onclick = async () => { // 通話が終了したときの処理 room.on('close', () => { console.log('通話が終了しました'); }); }; };
